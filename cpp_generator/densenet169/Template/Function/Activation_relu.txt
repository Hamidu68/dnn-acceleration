void SW_{Name}(DATA_T I[{Input_channel}][{Input_width}][{Input_height}], DATA_T O[{Output_channel}][{Output_width}][{Output_height}]) {{
	int m, x, y, i, j, k;
	for (m = 0; m<{Output_channel}; m++) {{
		for (x = 0; x<{Output_width}; x++) {{
			for (y = 0; y<{Output_height}; y++) {{
				if (I[m][x][y] < 0)
					O[m][x][y] = 0;
				else
				    O[m][x][y] = I[m][x][y];
			}}
		}}
	}}
}}
