void SW_{Name}(DATA_T I[{Input_channel}][{Input_width}][{Input_height}], DATA_T O[{Output_channel}][{Output_width}][{Output_height}])
{{
	int m, x, y;
	DATA_T ifm;
	DATA_T mv = {max_value};
	for (m = 0; m<{Input_channel}; m++) {{
		for (x = 0; x<{Input_width}; x++) {{
			for (y = 0; y<{Input_height}; y++) {{
				ifm = I[m][x][y];
				if (ifm >= mv)
					O[m][x][y] = mv;
				else if(ifm>=0.0 && ifm<mv)
					O[m][x][y] = ifm;
				else
				    O[m][x][y] = 0.0*(ifm-0.0);
			}}
		}}
	}}
}}
