void SW_{Name}(DATA_T I[{Input_channel}][{Input_width}][{Input_height}], DATA_T O[{Output_channel}][{Output_width}][{Output_height}], DATA_T W[{Output_channel}][{Filter_width}][{Filter_height}]) {{
	int m, x, y, i, j, k;
	DATA_T ifm, ofm;
	for (m = 0; m<{Output_channel}; m++) {{
		for (x = 0; x<{Output_width}; x++) {{
			for (y = 0; y<{Output_height}; y++) {{
				ofm = 0;
				for (i = 0; i<{Filter_width}; i++) {{
					for (j = 0; j<{Filter_height}; j++) {{
						if (x + i <= {Input_width} && y + j <= {Input_height}) {{
							ifm = I[m][x*{Stride_width} + i][y*{Stride_height} + j];
						}}

						ofm = ofm + ifm * W[m][i][j];
					}}
				}}
				O[m][x][y] = ofm;
			}}
		}}
	}}
}}
