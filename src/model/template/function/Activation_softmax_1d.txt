void SW_{Name}(DATA_T I[{Input_channel}], DATA_T O[{Output_channel}]) {{
	int i;
	DATA_T maximum;
	DATA_T denom = I[0];
	for (i = 1; i < {Input_channel}; i++)
		denom += I[i];
	for (i = 0; i < {Input_channel}; i++){{
		O[i] = I[i] / denom; //float
		if(i==0)
		    maximum=O[i];
		else{{
		    if(maximum<O[i])
		        maximum=O[i];

		}}
	}}
	for (i = 0; i < {Output_channel}; i++){{
	    if(maximum!=O[i])
	        O[i]=0;
	    else
	        O[i]=1;
	}}

}}
